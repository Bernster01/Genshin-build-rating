<script>
let dragging = false;
var selectedCharacter = null;
function selectCharacter(character){

if(dragging == false )
{


var img = document.getElementById("SelectCharcterImg");
var text = document.getElementById("SelectCharcterText");
img.src = "Characters/"+character+".png";
text.innerHTML = character;
text.
displayCharcters();
document.getElementById("LoadoutContainer").style.display = "inline";

}

}
function displayCharcters(){
var doc = document.getElementById("CharacterContainer");
var doc2 = document.getElementById("SelectCharcter");
if(doc.style.display == "none" || doc.style.display == "")
{
doc.style.display = "inline";
doc2.style.display = "none";

}
else
{
doc.style.display = "none";
doc2.style.display = "";

}

}


</script>
<script> //By ToddWebDev at https://codepen.io/toddwebdev/pen/yExKoj (modifed)





const slider = document.querySelector('.CharacterBox');
let isDown = false;
let startX;
let scrollLeft;
let update = 0;

slider.addEventListener('mousedown', (e) => {
  isDown = true;
  slider.classList.add('active');
  startX = e.pageX - slider.offsetLeft;
  scrollLeft = slider.scrollLeft;
});
slider.addEventListener('mouseleave', () => {
  isDown = false;
  slider.classList.remove('active');
  dragging = false;
});
slider.addEventListener('mouseup', () => {
  isDown = false;
  slider.classList.remove('active');
  dragging = false;
});
slider.addEventListener('mousemove', (e) => {
  if(!isDown) return;
  e.preventDefault();
  const x = e.pageX - slider.offsetLeft;
  const walk = (x - startX)
  slider.scrollLeft = scrollLeft - walk;
  dragging = true;
  
  

  });
 
  

</script>
